drop schema if exists vagtplantestdb;
CREATE SCHEMA `vagtplantestdb` DEFAULT CHARACTER SET utf8;

use vagtplantestdb;

drop table if exists medarbejdere;
CREATE TABLE `medarbejdere` (
  `username` varchar(70) NOT NULL,
  `password` varchar(255) NULL,
  `role` varchar(70) NOT NULL,
  `navn` varchar(70) NULL,
  `initialer` varchar(5) NULL,
  `telefonnummer` varchar(20) NULL,
  `visivagtplan` varchar(70) NULL,
  `medarbejderstatus` int NULL,
  `uselog` varchar(4500) NULL,
  `vagtansvarligsemail` varchar(70) NOT NULL,
  PRIMARY KEY (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

drop table if exists forbehold;
CREATE TABLE `forbehold` (
  `forbeholdid` int NOT NULL AUTO_INCREMENT,
  `dato` date NULL,
  `kommentar` varchar(450) NULL,
  `fk_username_forbehold` varchar(70) NOT NULL,
  PRIMARY KEY (`forbeholdid`),
  KEY `fk_username_idx` (`fk_username_forbehold`),
  CONSTRAINT `fk_username_forbehold` FOREIGN KEY (`fk_username_forbehold`) REFERENCES `medarbejdere` (`username`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

drop table if exists vagtbehov;
CREATE TABLE `vagtbehov` (
  `vagtbehovid` int NOT NULL auto_increment,
  `fra` datetime NOT NULL,
  `til` datetime NOT NULL,
  `titel` varchar(45) NOT NULL,
  `kommentar` varchar(450) NULL,
  PRIMARY KEY (`vagtbehovid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

drop table if exists vagtplaner;
CREATE TABLE `vagtplaner` (
  `vagtplanid` int NOT NULL auto_increment,
  `titel` varchar(45) NOT NULL,
  `kommentar` varchar(450) NULL,
  `fra` datetime NOT NULL,
  `til` datetime NOT NULL,
  PRIMARY KEY (`vagtplanid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

drop table if exists vagter;
CREATE TABLE `vagter` (
  `vagtid` int NOT NULL auto_increment,
  `fk_username` varchar(70) NOT NULL,
  `fra` datetime NOT NULL,
  `til` datetime NOT NULL,
  `titel` varchar(45) NULL,
  `kommentar` varchar(450) NULL,
  PRIMARY KEY (`vagtid`),
  KEY `fk_username_idx` (`fk_username`),
  CONSTRAINT `fk_username` FOREIGN KEY (`fk_username`) REFERENCES `medarbejdere` (`username`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
